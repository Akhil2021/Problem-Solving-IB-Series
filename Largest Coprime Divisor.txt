public class Solution {
    public int gcd(int A,int B)
    {
        if(B==0)
            return A;
        return gcd(B,A%B);
    }
    public int cpFact(int A, int B) {
        int gcd = gcd(A,B);
        while(gcd!=1)
        {
            A=A/gcd;
            gcd=gcd(A,B);
        }
        return A;
    }
}
