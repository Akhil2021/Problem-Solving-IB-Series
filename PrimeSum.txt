public class Solution {
    public int[] primesum(int A) {
        for(int i=2;i<A/2;i++)
        {
            int num1 = i;
            int num2 = A -i;
            int count1=0,count2=0;
            for(int j=2;j<num1;j++)
            {
                if(num1%j==0)
                {
                    count1++;
                }
            }
        }
    }
}
